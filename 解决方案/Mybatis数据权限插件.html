Mybatis数据权限插件

插件只实现了单表sql的拦截, 对于多表join的使用@DataScope控制(多表用插件控制太麻烦了! bug感觉也会很多)

​

思路是参考ruoyi的@DataScope的实现. 在sql中注入一段代码达到过滤数据权限的效果. 例如:

select nick_name,user_name from user_order where nick_name like xx
@DataScope会转变为
select nick_name,user_name 
from 
    user_order
  left join sys_user on sys_user.id = user_order.id 
where
    nick_name like xx
    and sys_user.dept_id in (SELECT dept_id FROM sys_user WHERE dept_id = XX or find_in_set( XX , ancestors ))

可以看到, 有3个操作

  1. 连sys_user表
  2. 子查询获取用户所在部门以及下属部门的id
  3. 判断user_order数据对应用户所在部门是否在子查询中

以上三点达到一个效果: 查询所在部门以及下属部门的nick_name为xx的user_order行

这是以部门为权限隔离的逻辑, 还能更细一点, 具体实现也是类似的

具体实现

实现插件

@Data
@NoArgsConstructor
@AllArgsConstructor
@ToString(callSuper = true)
@EqualsAndHashCode(callSuper = true)
@SuppressWarnings({"rawtypes"})
public class DataScopeInterceptor extends JsqlParserSupport implements InnerInterceptor {

      // 需要不做权限控制的表
    private List<String> ignoreTableNames = ImmutableList.of(
            "gen_table"
    );

    // 表对应的所属userId. 因为有些表userId列名不一定是userId
    private Map<String, String> tableUserIdColumnMap = new HashMap<String, String>() {{
        put("xy_psychological_assessment_reservation", "reservation_id");
        put("xy_psychological_assessment_consult", "processors_id");
    }};

    @Override
    public void beforeQuery(Executor executor, MappedStatement ms, Object parameter, RowBounds rowBounds, ResultHandler resultHandler, BoundSql boundSql) throws SQLException {
        // 不清楚这个方法是干嘛的...但parserSingle会回调processSelect
        if (InterceptorIgnoreHelper.willIgnoreDataPermission(ms.getId())) return;
        PluginUtils.MPBoundSql mpBs = PluginUtils.mpBoundSql(boundSql);
        mpBs.sql(parserSingle(mpBs.sql(), ms.getId()));
    }

    @Override
    protected void processSelect(Select select, int index, String sql, Object obj) {
        if (!SecurityContextHolder.getDataScope()) {
            // 从上下文中判断是否开启权限拦截, 因为默认是没有的. 贸然全部开启可能会有很多bug
            return;
        }

        SelectBody selectBody = select.getSelectBody();
        if (selectBody instanceof PlainSelect) {
            this.setWhere((PlainSelect) selectBody, (String) obj);
        } else if (selectBody instanceof SetOperationList) {
            SetOperationList setOperationList = (SetOperationList) selectBody;
            List<SelectBody> selectBodyList = setOperationList.getSelects();
            selectBodyList.forEach(s -> this.setWhere((PlainSelect) s, (String) obj));
        }
    }

    protected void setWhere(PlainSelect plainSelect, String whereSegment) {
        try {
            this.setWhere0(plainSelect, whereSegment);
        } catch (Exception e) {
            logger.error("DataScopeInterceptor setWhere error", e);
        }
    }

    /**
     * 设置 where 条件
     *
     * @param plainSelect  查询对象
     * @param whereSegment 查询条件片段
     */
    protected void setWhere0(PlainSelect plainSelect, String whereSegment) throws Exception {
        Table fromTable;
        FromItem fromItem = plainSelect.getFromItem();
        if (fromItem instanceof Table && CollectionUtils.isEmpty(plainSelect.getJoins())) {
            /**
             * 只处理单表, 多表xml的情况下自己关联user表并注入${params.dataScope}
             * {@link com.ruoyi.common.datascope.annotation.DataScope}
             */
            fromTable = (Table) fromItem;
        } else {
            return;
        }
        String tableName = fromTable.getName();;
        if (ignoreTableNames.contains(tableName)) {
            return;
        }

        LoginUser loginUser = SecurityUtils.getLoginUser();
        SysUser sysUser;
        Long deptId;
        if (loginUser == null || (sysUser =loginUser.getSysUser()) == null ||
                (deptId = sysUser.getDeptId()) == null || deptId == 0) {
            // 拿不到部门 跳过
            return;
        }
        if (sysUser.getRoles().stream().anyMatch(m -> "1".equals(m.getDataScope()) ||
                "超级管理员".equals(m.getRoleType()) ||
                "supperAdmin".equals(m.getRoleKey()))) {
            // 所属角色有一个是数据权限是所有的 或者超级管理员 跳过
            // 1: 所有数据权限 4:所在部门及下属部门数据权限
            return;
        }

        String userIdColumn = tableUserIdColumnMap.getOrDefault(tableName, "user_id");

        // 处理查询字段名 => 加上表前缀
        plainSelect.getSelectItems().forEach(m -> {
            Expression expression = ((SelectExpressionItem) m).getExpression();
            if (expression instanceof Column) {
                Column column = (Column) expression;
                column.setTable(fromTable);
            } else if (expression instanceof Function) {
                // 函数类型(count(*)) 跳过
            }
        });


        // join
        if (!"sys_user".equals(tableName)) {
            Join userJoin = new Join();
            userJoin.setOnExpressions(Collections.singletonList(new EqualsTo(new Column(tableName + "." + userIdColumn), new Column("sys_user.user_id"))));
            userJoin.setLeft(true);
            userJoin.setRightItem(new Table("sys_user"));
            plainSelect.setJoins(Collections.singletonList(userJoin));
        }


        // where
        Expression where = plainSelect.getWhere();
        handlerWhere(where, fromTable);

        // sys_user.dept_id in (select dept_id from sys_dept where dept_id = 102 or find_in_set( 102 , ancestors ))
        String query = String.format("SELECT dept_id FROM sys_dept WHERE dept_id = %s or find_in_set( %s , ancestors )", deptId, deptId);
        Select subSelect = (Select) CCJSqlParserUtil.parse(query);
        SubSelect subSelectNode = new SubSelect();
        subSelectNode.setSelectBody(subSelect.getSelectBody());

        // 填充sql
        InExpression equalsTo = new InExpression();
        equalsTo.setLeftExpression(new Column("sys_user.dept_id"));
        equalsTo.setRightExpression(subSelectNode);
        if (where == null) {
            plainSelect.setWhere(equalsTo);
        } else {
            plainSelect.setWhere(new AndExpression(where, equalsTo));
        }

    }

    private void handlerWhere(Object where, Table table) throws Exception {
        if (where == null) {
            return;
        }

        if (where instanceof Column) {
            Column column = (Column) where;
            // Column只需要setTable就可以加表名, 如果有别名就是d.dept_id
            // 但是应该是有bug, 理论上不用这一个判断
            if (column.getColumnName().contains(".")) {
                // 已经是带表名的字段 dept.dept_id
                return;
            }
            column.setTable(table);
            return;
        }

        if (!(where instanceof Expression)) {
            return;
        }

        if (where instanceof Parenthesis) {
            // (test_id = ? AND user_id = ?) 这种表达式
            where = ((Parenthesis) where).getExpression();
        }

        /*
        这里是核心 大部分都是AndExpression, AndExpression会有Left,Right2个Expression
        如果存在多个表达式如: id=1 and name like x. 那对应Left就是id=1
        如果只存在1个表达式如: id=1. 那对应Left就是id(对应类为Column)
        所以只需要递归Left和Right将类型为Column的设置表名就可以
        */
        Class<?> aClass = where.getClass();
        Optional<Method> getLeftExpression = Arrays.stream(aClass.getMethods())
                .filter(f -> f.getName().equals("getLeftExpression"))
                .filter(method -> method.getParameterCount() == 0)
                .findFirst();
        if (getLeftExpression.isPresent()) {
            // 尝试获取指定名称的属性
            Method method = getLeftExpression.get();
            Object invoke = method.invoke(where);
            // 设置属性可访问，即使是 private 属性也可以访问
//            field.setAccessible(true);

            handlerWhere(invoke, table);
        }

        Optional<Method> getRightExpression = Arrays.stream(aClass.getMethods())
                .filter(f -> f.getName().equals("getRightExpression"))
                .filter(method -> method.getParameterCount() == 0)
                .findFirst();
        if (getRightExpression.isPresent()) {
            Method method = getRightExpression.get();
            Object invoke = method.invoke(where);
            // 设置属性可访问，即使是 private 属性也可以访问
//            field.setAccessible(true);

            handlerWhere(invoke, table);
        }

    }


}

注入插件

@EnableTransactionManagement(proxyTargetClass = true)
@Configuration
public class MybatisPlusConfig {
    @Bean
    public MybatisPlusInterceptor mybatisPlusInterceptor() {
        MybatisPlusInterceptor interceptor = new MybatisPlusInterceptor();
        // 多租户插件
        interceptor.addInnerInterceptor(new TenantLineInnerInterceptor(new SystemTenantLineHandler()));
        // 数据隔离插件 => 替换为自己的插件
        interceptor.addInnerInterceptor(new DataScopeInterceptor());
        return interceptor;
    }
}

多表数据隔离

理论上插件也可以做到多表数据隔离, 但是现在因为已经有很多业务了. 贸然全部都加上可能会有很多bug

​

多表数据隔离是若依框架中使用注解方式实现的, 具体逻辑是

  1. 参数必须实现对应BaseEntity(需要有特定属性值注入)
  2. 连表查询操作需要自己连上对应的数据表
  3. 在xml中注入sql

实体类

主要是需要params属性

@Data
@Accessors(chain = true)
public class BasePage<T> {
    /**
     * 分页页码
     */
    private Integer pageNum = 1;
    /**
     * 分页大小
     */
    private Integer pageSize = 20;

    @JsonInclude(JsonInclude.Include.NON_EMPTY)
    @TableField(exist = false)
    private Map<String, Object> params;

    public IPage<T> getPage() {
        return new Page<T>(pageNum, pageSize);
    }

    public Map<String, Object> getParams() {
        if (params == null) {
            params = new HashMap<>();
        }
        return params;
    }
}

具体切面

具体逻辑依旧是注入一个sql, 不过这里是注入到实体的params属性值里

@Aspect
@Component
public class DataScopeAspect {
    /**
     * 全部数据权限
     */
    public static final String DATA_SCOPE_ALL = "1";

    /**
     * 自定数据权限
     */
    public static final String DATA_SCOPE_CUSTOM = "2";

    /**
     * 部门数据权限
     */
    public static final String DATA_SCOPE_DEPT = "3";

    /**
     * 部门及以下数据权限
     */
    public static final String DATA_SCOPE_DEPT_AND_CHILD = "4";

    /**
     * 仅本人数据权限
     */
    public static final String DATA_SCOPE_SELF = "5";

    /**
     * 数据权限过滤关键字
     */
    public static final String DATA_SCOPE = "dataScope";

    @Before("@annotation(dataScope)")
    public void doBefore(JoinPoint point, DataScope dataScope) {
        // 上下文控制是否开启
        if (SecurityContextHolder.getDataScope() || clearDataScope(point)) {
            handleDataScope(point, dataScope);
        }
    }

    /**
     * 拼接权限sql前先清空params.dataScope参数防止注入
     * BaseEntity等支持的类需要存在params属性
     */
    private boolean clearDataScope(final JoinPoint joinPoint) {
        Object[] args = joinPoint.getArgs();
        boolean hasDataScopeColumn = false;
        for (Object params : args) {

            if (StringUtils.isNotNull(params) && params instanceof BaseEntity) {
                BaseEntity baseEntity = (BaseEntity) params;
                baseEntity.getParams().put(DATA_SCOPE, "");
                hasDataScopeColumn = true;
            } else if (params instanceof BaseDomain) {
                BaseDomain baseEntity = (BaseDomain) params;
                baseEntity.getParams().put(DATA_SCOPE, "");
                hasDataScopeColumn = true;
            } else if (params instanceof BasePage) {
                BasePage basePage = (BasePage) params;
                basePage.getParams().put(DATA_SCOPE, "");
                hasDataScopeColumn = true;
            }

        }

        return hasDataScopeColumn;

    }

    protected void handleDataScope(final JoinPoint joinPoint, DataScope controllerDataScope) {
        // 获取当前的用户
        LoginUser loginUser = SecurityUtils.getLoginUser();
        if (StringUtils.isNotNull(loginUser)) {
            SysUser currentUser = loginUser.getSysUser();
            // 如果是超级管理员，则不过滤数据
            if (StringUtils.isNotNull(currentUser) &&
                    currentUser.getRoles().stream().noneMatch(sysRole ->
                            "超级管理员".equals(sysRole.getRoleType()) || DATA_SCOPE_ALL.equals(sysRole.getDataScope()))
            ) {

                String permission = StringUtils.defaultIfEmpty(controllerDataScope.permission(), SecurityContextHolder.getPermission());
                dataScopeFilter(joinPoint, currentUser, controllerDataScope.deptAlias(), controllerDataScope.userAlias(), permission);
            }
        }
    }

    /**
     * 数据范围过滤
     *
     * @param joinPoint  切点
     * @param user       用户
     * @param deptAlias  部门别名
     * @param userAlias  用户别名
     * @param permission 权限字符
     */
    public static void dataScopeFilter(JoinPoint joinPoint, SysUser user, String deptAlias, String userAlias, String permission) {
        StringBuilder sqlString = new StringBuilder();
        List<String> conditions = new ArrayList<String>();

        for (SysRole role : user.getRoles()) {
            String dataScope = role.getDataScope();
            if (!DATA_SCOPE_CUSTOM.equals(dataScope) && conditions.contains(dataScope)) {
                continue;
            }
            if (StringUtils.isNotEmpty(permission) && StringUtils.isNotEmpty(role.getPermissions())
                    && !StringUtils.containsAny(role.getPermissions(), Convert.toStrArray(permission))) {
                continue;
            }
            if (DATA_SCOPE_ALL.equals(dataScope)) {
                sqlString = new StringBuilder();
                conditions.add(dataScope);
                break;
            } else if (DATA_SCOPE_CUSTOM.equals(dataScope)) {
                sqlString.append(StringUtils.format(
                        " OR {}.dept_id IN ( SELECT dept_id FROM sys_role_dept WHERE role_id = {} ) ", deptAlias,
                        role.getRoleId()));
            } else if (DATA_SCOPE_DEPT.equals(dataScope)) {
                sqlString.append(StringUtils.format(" OR {}.dept_id = {} ", deptAlias, user.getDeptId()));
            } else if (DATA_SCOPE_DEPT_AND_CHILD.equals(dataScope)) {
                sqlString.append(StringUtils.format(
                        " OR {}.dept_id IN ( SELECT dept_id FROM sys_dept WHERE dept_id = {} or find_in_set( {} , ancestors ) )",
                        deptAlias, user.getDeptId(), user.getDeptId()));
            } else if (DATA_SCOPE_SELF.equals(dataScope)) {
                if (StringUtils.isNotBlank(userAlias)) {
                    sqlString.append(StringUtils.format(" OR {}.user_id = {} ", userAlias, user.getUserId()));
                } else {
                    // 数据权限为仅本人且没有userAlias别名不查询任何数据
                    sqlString.append(StringUtils.format(" OR {}.dept_id = 0 ", deptAlias));
                }
            }
            conditions.add(dataScope);
        }

        // 多角色情况下，所有角色都不包含传递过来的权限字符，这个时候sqlString也会为空，所以要限制一下,不查询任何数据
        if (StringUtils.isEmpty(conditions)) {
            sqlString.append(StringUtils.format(" OR {}.dept_id = 0 ", deptAlias));
        }

        if (StringUtils.isNotBlank(sqlString.toString())) {
            Object[] args = joinPoint.getArgs();
            String dataScopeSql = " AND (" + sqlString.substring(4) + ")";
            for (Object params : args) {

                if (params instanceof BaseEntity) {
                    ((BaseEntity) params).getParams().put(DATA_SCOPE, dataScopeSql);
                } else if (params instanceof BaseDomain) {
                    ((BaseDomain) params).getParams().put(DATA_SCOPE, dataScopeSql);
                } else if (params instanceof BasePage) {
                    ((BasePage) params).getParams().put(DATA_SCOPE, dataScopeSql);
                }

            }
        }
    }
}

        select distinct u.user_id, u.dept_id, u.user_name, u.nick_name, u.email, u.phonenumber, u.status, u.create_time
        from sys_user u
        left join sys_dept d on u.dept_id = d.dept_id
        left join sys_user_role ur on u.user_id = ur.user_id
        left join sys_role r on r.role_id = ur.role_id
        where u.del_flag = '0' and r.role_id = #{roleId} and r.tenant_id = #{tenantId}
        <if test="userName != null and userName != ''">
            AND u.user_name like concat('%', #{userName}, '%')
        </if>
        <if test="phonenumber != null and phonenumber != ''">
            AND u.phonenumber like concat('%', #{phonenumber}, '%')
        </if>
        <!-- 数据范围过滤 -->
        ${params.dataScope}
    </select>
