文件分片上传

SpringBoot实现大文件上传/下载(分片、断点续传)

entity

@Data
public class UploadFileInfo {
    private MultipartFile file;

    /**
     * 文件名称
     */
    private String fileName;

    /**
     * 文件md5
     */
    private String md5;

    /**
     * 上传文件会有多个分片，记录当前为那个分片
     */
    private Integer currentChunk;

    /**
     * 总分片数
     */
    private Integer chunks;

    /**
     * {md5_fileName}列表
     * 当currentChunk = chunks时需要传 并且有序 名字可以自定义, 只要能对得上就可以
     */
    private List<String> md5FileNameList;

    public UploadFileInfo(HttpServletRequest request, MultipartFile file) {
        this.file = file;
        this.fileName = request.getParameter("fileName");
        if (fileName == null || "".equals(fileName)) {
            fileName = file.getOriginalFilename();
        }
        this.md5 = request.getParameter("md5");
        this.currentChunk = Integer.parseInt(request.getParameter("currentChunk"));
        this.chunks = Integer.parseInt(request.getParameter("chunks"));
        String md5FileNameListStr = request.getParameter("md5FileNameList");
        if (md5FileNameListStr != null && !"".equals(md5FileNameListStr)) {
            this.md5FileNameList = JSONUtil.parseArray(md5FileNameListStr).toList(String.class);
        }

        if (chunks == null || chunks == 0 ||
                currentChunk == null || currentChunk == 0) {
            throw new ServiceException("分片参数异常");
        }

        if (md5 == null || "".equals(md5)) {
            throw new ServiceException("文件md5异常");
        }

        if (fileName == null || "".equals(fileName)) {
            throw new ServiceException("文件异常");
        }

        if (chunks.intValue() == currentChunk && md5FileNameList.isEmpty()) {
            throw new ServiceException("参数异常");
        }
    }

}

@PostMapping("checkChunkStatus")
public AjaxResult checkChunkStatus(HttpServletRequest request, MultipartFile file) throws Exception {
    UploadFileInfo uploadFileInfo = new UploadFileInfo(request, file);
    String md5 = uploadFileInfo.getMd5();

    String tempFileMd5 = chunkFileService.getTempFileMd5(uploadFileInfo);

service

@Service
public class ChunkFileService {

    /**
     * 上传文件存储在本地的根路径
     * /project/xingyun/file/fileData
     */
    @Value("${file.path}")
    private String uploadPath;

    @Getter
    private String tempPath = System.getProperty("user.dir") + "/temp";

    /**
     * 编码
     */
    private static final String UTF_8 = "UTF-8";

    /**
     * 文件上传路径(当前项目路径下，也可配置固定路径)
     */
//    private String uploadPath = System.getProperty("user.dir") + "/springboot-file/upload/";

    /**
     * 文件下载地址(当前项目路径下，也可配置固定路径)
     */
    private String downloadPath = System.getProperty("user.dir") + "/springboot-file/download/";

    /**
     * 分片下载每一片大小为50M
     */
    private static final Long PER_SLICE = 1024 * 1024 * 50L;

    /**
     * 定义分片下载线程池
     */
//    private ExecutorService executorService = Executors.newFixedThreadPool(Runtime.getRuntime().availableProcessors());

    /**
     * final string
     */
    private static final String RANGE = "Range";

    @PostConstruct
    public void init() {
//        tempPath = uploadPath + File.separator + "temp";
    }

    /**
     * 迁移临时文件到文件目录
     *
     * @param tempPath 临时文件路径
     * @param fileName 文件名
     * @return 文件全路径
     * @throws Exception e
     */
    public String transferTempFile(String tempPath, String fileName) throws Exception {
        File tempFile = new File(tempPath);
        File targetFile = new File(uploadPath + XyFileUtils.datePath() + fileName);

        if (!tempFile.exists()) {
            throw new ServiceException("文件不存在");
        }

        if (!targetFile.getParentFile().exists()) {
            targetFile.getParentFile().mkdirs();
        }

        targetFile.delete();

        try (FileInputStream fisMap = new FileInputStream(tempFile);
             FileOutputStream fos = new FileOutputStream(targetFile)) {

            FileChannel channel = fisMap.getChannel();
            channel.transferTo(0, channel.size(), fos.getChannel());
        }

        return targetFile.getAbsolutePath();
    }

    /**
     * 分片文件的命名规范
     *
     * @param uploadFileInfo uploadFileInfo
     * @return 临时文件命名
     */
    public String getTempFileName(UploadFileInfo uploadFileInfo) {
        return uploadFileInfo.getCurrentChunk() + "_" + uploadFileInfo.getMd5() + "_" + uploadFileInfo.getFileName();
    }

    /**
     * 分片文件的存放绝对路径
     *
     * @param uploadFileInfo uploadFileInfo
     * @return 临时文件全路径
     */
    public String getTempFilePath(UploadFileInfo uploadFileInfo) {
        return XyFileUtils.join(tempPath, getTempFileName(uploadFileInfo));
    }

    public String getTempFileMd5(UploadFileInfo uploadFileInfo) throws Exception {
        File file = new File(getTempFilePath(uploadFileInfo));
        if (!file.exists()) {
            return StringUtils.EMPTY;
        }
        return XyFileUtils.getFileMD5(file.toString());
    }

    /**
     * 文件分片上传
     *
     * @return 分片上传完成 返回文件全路径名 否则返回null
     */
    public String uploadFileChunk(UploadFileInfo uploadFileInfo, boolean isMerge) throws Exception {
        // 写入临时文件
        writeTempFile(uploadFileInfo);
        // 判断是否合并(完成)
        return mergeFile(uploadFileInfo, isMerge);
    }

    /**
     * 获取ServletFileUpload
     */
    private ServletFileUpload getServletFileUpload() {
        // 设置缓冲区大小，先读到内存里在从内存写
        DiskFileItemFactory factory = new DiskFileItemFactory();
        factory.setSizeThreshold(1024);
        File file = new File(uploadPath);
        // 如果文件夹不存在则创建
        if (!file.exists() && !file.isDirectory()) {
            file.mkdirs();
        }
        factory.setRepository(file);
        // 解析
        ServletFileUpload upload = new ServletFileUpload(factory);
        // 设置单个大小与最大大小
        upload.setFileSizeMax(1 * 1024 * 1024 * 1024L);
        upload.setSizeMax(10 * 1024 * 1024 * 1024L);
        return upload;
    }

    /**
     * 写入临时文件
     *
     * @param items
     * @param uploadFileInfo
     * @throws Exception
     */
    private void writeTempFile(UploadFileInfo uploadFileInfo) throws Exception {
        // 有分片需要临时目录
        String tempFileName = getTempFileName(uploadFileInfo);
        File tempFile = new File(tempPath, tempFileName);

        // 判断文件是否存在 => 存在则比较是否相同
        if (tempFile.exists() &&
                uploadFileInfo.getFile().getSize() == tempFile.length()) {
            // TODO 加上md5校验
            return;
        }

        if (!tempFile.getParentFile().exists()) {
            tempFile.getParentFile().mkdirs();
        }

        uploadFileInfo.getFile().transferTo(tempFile);
    }

    /**
     * 判断是否合并
     *
     * @param uploadFileInfo
     * @param isMerge
     * @throws IOException
     * @throws InterruptedException
     */
    private String mergeFile(UploadFileInfo uploadFileInfo, boolean isMerge) throws IOException, InterruptedException {
        Integer currentChunk = uploadFileInfo.getCurrentChunk();
        Integer chunks = uploadFileInfo.getChunks();
        String fileName = uploadFileInfo.getMd5() + "_" + uploadFileInfo.getFileName();
        List<String> md5FileNameList = uploadFileInfo.getMd5FileNameList();
        // 如果当前分片等于总分片那么合并文件
        if (currentChunk != null && chunks != null && currentChunk.intValue() == chunks) {

            if (md5FileNameList.size() != currentChunk) {
                throw new ServiceException("文件分片不完整");
            }

            File tempFile = new File(tempPath, fileName);
            try (FileOutputStream fos = new FileOutputStream(tempFile)) {
                // 根据之前命名规则找到所有分片

                for (int i = 0; i < md5FileNameList.size(); i++) {
                    String md5File = md5FileNameList.get(i);
                    File file = new File(tempPath, (i + 1) + "_" + md5File);

                    // 并发情况，需要判断所有，因为可能最后一个分片传完，之前有的还没传完
                    while (!file.exists()) {
                        // 不存在休眠100毫秒后在重新判断
                        Thread.sleep(100);
                    }

                    if (!isMerge) {
                        continue;
                    }

                    try (FileInputStream fileInputStream = new FileInputStream(file)) {

                        FileChannel channel = fileInputStream.getChannel();
                        channel.transferTo(0, channel.size(), fos.getChannel());

    private DownloadFileInfo getDownloadFileInfo(long fSize, HttpServletRequest request, HttpServletResponse response) {
        long pos = 0;
        long last = fSize - 1;
        // 判断前端是否需要分片下载
        if (request.getHeader(RANGE) != null) {
            response.setStatus(HttpServletResponse.SC_PARTIAL_CONTENT);
            String numRange = request.getHeader(RANGE).replace("bytes=", "");
            String[] strRange = numRange.split("-");
            if (strRange.length == 2) {
                pos = Long.parseLong(strRange[0].trim());
                last = Long.parseLong(strRange[1].trim());
                // 若结束字节超出文件大小，取文件大小
                if (last > fSize - 1) {
                    last = fSize - 1;
                }
            } else {
                // 若只给一个长度，开始位置一直到结束
                pos = Long.parseLong(numRange.replace("-", "").trim());
            }
        }
        long rangeLength = last - pos + 1;
        String contentRange = "bytes " + pos + "-" + last + "/" + fSize;
        return new DownloadFileInfo(fSize, pos, last, rangeLength, contentRange);
    }

    private void mergeFile(String fName, long page) throws IOException {
        File file = new File(downloadPath, fName);

        try (BufferedOutputStream os = new BufferedOutputStream(new FileOutputStream(file))) {
            for (int i = 0; i <= page; i++) {
                File tempFile = new File(downloadPath, i + "-" + fName);
                // 文件不存在或文件没写完
                while (!tempFile.exists() || (i != page && tempFile.length() < PER_SLICE)) {
                    Thread.sleep(100);
                }
                byte[] bytes = FileUtils.readFileToByteArray(tempFile);
                os.write(bytes);
                os.flush();
                tempFile.delete();
            }
            // 删除文件
            File f = new File(downloadPath, "-1" + "-null");
            if (f.exists()) {
                f.delete();
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    /**
     * 设置响应头
     */
    private void setResponse(HttpServletResponse response, String fileName, DownloadFileInfo downloadFileInfo) throws UnsupportedEncodingException {
        response.setCharacterEncoding(UTF_8);
        response.setContentType("application/x-download");
        response.addHeader("Content-Disposition", "attachment;filename=" + URLEncoder.encode(fileName, UTF_8));
        // 支持分片下载
        response.setHeader("Accept-Range", "bytes");
        response.setHeader("fSize", String.valueOf(downloadFileInfo.getFSize()));
        response.setHeader("fName", URLEncoder.encode(fileName, UTF_8));
        // range响应头
        response.setHeader("Content-Range", downloadFileInfo.getContentRange());
        response.setHeader("Content-Length", String.valueOf(downloadFileInfo.getRangeLength()));
    }

@NoArgsConstructor(access = AccessLevel.PRIVATE)
public class XyFileUtils {

    public static String datePath() {
        return DateFormatUtils.format(new Date(), "/yyyy/MM/dd/");
    }

    /**
     * 拼接路径
     *
     * @param paths 路径
     * @return 拼接后的路径
     */
    public static String join(String basePath, String... paths) {
        return Paths.get(basePath, paths).toString();
    }

    /**
     * 获取父目录
     *
     * @param filePath 目录 /xy/xx/xxx.pdf
     * @return /xy/xx
     */
    public static String getParentUri(String filePath) {
        return Paths.get(filePath).getParent().toString();
    }

    public static String getFileName(String filePath) {
        return FileNameUtil.getName(filePath);
    }

    public static String getFileNameSimple(String filePath) {
        return FileNameUtil.getPrefix(filePath);
    }

    /**
     * 修改文件后缀
     * /xy/2024/03/24/12/xxx.pfd => /xy/2024/03/24/12/xxx.word
     *
     * @param filePath 文件全路径 需要文件
     * @param suffix   新后缀
     * @return 修改后文件路径
     */
    public static String changeSuffix(String filePath, String suffix) {
        return join(getParentUri(filePath), FileNameUtil.getPrefix(filePath) + "." + suffix);
    }

    public static String getFileMD5(String filePath) throws IOException, NoSuchAlgorithmException {
        MessageDigest md = MessageDigest.getInstance("MD5");
        try (FileInputStream fis = new FileInputStream(filePath);
             DigestInputStream dis = new DigestInputStream(fis, md)) {
            byte[] buffer = new byte[4096];
            while (dis.read(buffer) != -1) {
                // Read file stream and update digest
            }
            md = dis.getMessageDigest();
        }
        return bytesToHex(md.digest());
    }

    private static String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for (byte b : bytes) {
            sb.append(String.format("%02x", b));
        }
        return sb.toString();
    }
